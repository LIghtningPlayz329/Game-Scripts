print("made by suitdev")

local a1 = getfenv()["game"]
local b2, c3, d4 = {}, {}, {}

for _,e5 in pairs({"Players", "TweenService", "ReplicatedStorage", "CoreGui"}) do
	b2[#b2+1] = a1:GetService(e5)
end

local srvPlayers, srvTween, srvStorage, guiCore = unpack(b2)
local plr = srvPlayers.LocalPlayer

local function chr(...) return string.char(...) end
local function sleep(...) return task.wait(...) end
local function getGuiName() return chr(76,111,97,100,105,110,103,83,99,114,101,101,110,80,114,101,102,97,98) end

if not a1:IsLoaded() then a1.Loaded:Wait() end
repeat sleep() until plr.Character and plr:FindFirstChild("PlayerGui") and plr.PlayerGui:FindFirstChild(getGuiName()) == nil
srvStorage:WaitForChild("Remotes"):WaitForChild("EndDecision"):FireServer(false)

if not guiCore:FindFirstChild("bond_ui") then
	local g1 = Instance.new("ScreenGui", guiCore)
	g1.Name = "bond_ui"

	local frm = Instance.new("Frame")
	frm.Name = "main_frame"
	frm.Size = UDim2.new(0.13,0,0.1,0)
	frm.Position = UDim2.new(0.03,0,0.05,0)
	frm.BackgroundColor3 = Color3.new(1,1,1)
	frm.BackgroundTransparency = 0.3
	frm.BorderSizePixel = 1
	frm.BorderColor3 = Color3.new()
	frm.Draggable = true
	frm.Active = true
	frm.Parent = g1

	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0.23,0)
	corner.Parent = frm

	local stroke = Instance.new("UIStroke")
	stroke.Color = Color3.new(0,0,1)
	stroke.Thickness = 2
	stroke.Parent = frm

	local lbl = Instance.new("TextLabel")
	lbl.Name = "label"
	lbl.Size = UDim2.new(1,0,1,0)
	lbl.Text = "Starting..."
	lbl.FontFace = Font.new("rbxassetid://12187372175", Enum.FontWeight.Regular, Enum.FontStyle.Normal)
	lbl.TextSize = 20
	lbl.TextColor3 = Color3.new()
	lbl.BackgroundTransparency = 1
	lbl.Parent = frm
end

_G.bonds = 0

workspace.RuntimeItems.ChildAdded:Connect(function(v)
	if v.Name:find("Bond") and v:FindFirstChild("Part") then
		v.Destroying:Connect(function()
			_G.bonds += 1
		end)
	end
end)

spawn(function()
	repeat
		sleep()
		local gui = guiCore:FindFirstChild("bond_ui")
		if gui and gui:FindFirstChild("main_frame") and gui.main_frame:FindFirstChild("label") then
			gui.main_frame.label.Text = "Bond (+" .. tostring(_G.bonds) .. ")"
		end
	until not guiCore:FindFirstChild("bond_ui")
end)

plr.CameraMode = "Classic"
plr.CameraMaxZoomDistance = math.huge
plr.CameraMinZoomDistance = 30

local char = plr.Character
char:WaitForChild("HumanoidRootPart").Anchored = true
sleep(0.5)

repeat
	sleep(0.05)
	char.HumanoidRootPart.Anchored = true
	char.HumanoidRootPart.CFrame = CFrame.new(80, 3, -9000)
until workspace.RuntimeItems:FindFirstChild("MaximGun")

sleep(0.2)

for _,v in pairs(workspace.RuntimeItems:GetChildren()) do
	if v.Name == "MaximGun" and v:FindFirstChild("VehicleSeat") then
		v.VehicleSeat.Disabled = false
		v.VehicleSeat:SetAttribute("Disabled", false)
		v.VehicleSeat:Sit(char.Humanoid)
	end
end

sleep(0.3)

for _,v in pairs(workspace.RuntimeItems:GetChildren()) do
	if v.Name == "MaximGun" and v:FindFirstChild("VehicleSeat") and 
		(char.HumanoidRootPart.Position - v.VehicleSeat.Position).Magnitude < 400 then
		char.HumanoidRootPart.CFrame = v.VehicleSeat.CFrame
	end
end

sleep(0.9)
char.HumanoidRootPart.Anchored = false

repeat sleep() until char.Humanoid.Sit == true
sleep(0.5)
char.Humanoid.Sit = false
sleep(0.5)

repeat
	sleep()
	for _,v in pairs(workspace.RuntimeItems:GetChildren()) do
		if v.Name == "MaximGun" and v:FindFirstChild("VehicleSeat") and 
			(char.HumanoidRootPart.Position - v.VehicleSeat.Position).Magnitude < 400 then
			char.HumanoidRootPart.CFrame = v.VehicleSeat.CFrame
		end
	end
until char.Humanoid.Sit == true

sleep(1)

for _,v in pairs(workspace:GetChildren()) do
	if v:IsA("Model") and v:FindFirstChild("RequiredComponents") then
		local rc = v.RequiredComponents
		if rc:FindFirstChild("Controls") and rc.Controls:FindFirstChild("ConductorSeat") and rc.Controls.ConductorSeat:FindFirstChild("VehicleSeat") then
			local seat = rc.Controls.ConductorSeat.VehicleSeat
			local tween = srvTween:Create(char.HumanoidRootPart, TweenInfo.new(25, Enum.EasingStyle.Quad, Enum.EasingDirection.InOut), {
				CFrame = seat.CFrame * CFrame.new(0, 20, 0)
			})
			tween:Play()
			if not char.HumanoidRootPart:FindFirstChild("VelocityHandler") then
				local bv = Instance.new("BodyVelocity")
				bv.Name = "VelocityHandler"
				bv.Parent = char.HumanoidRootPart
				bv.MaxForce = Vector3.new(9e9,9e9,9e9)
				bv.Velocity = Vector3.zero
			end
			tween.Completed:Wait()
		end
	end
end

sleep(1)

while true do
	if char.Humanoid.Sit then
		local tween = srvTween:Create(char.HumanoidRootPart, TweenInfo.new(17, Enum.EasingStyle.Quad, Enum.EasingDirection.InOut), {
			CFrame = CFrame.new(0.5, -78, -49429)
		})
		tween:Play()

		if not char.HumanoidRootPart:FindFirstChild("VelocityHandler") then
			local bv = Instance.new("BodyVelocity")
			bv.Name = "VelocityHandler"
			bv.Parent = char.HumanoidRootPart
			bv.MaxForce = Vector3.new(9e9,9e9,9e9)
			bv.Velocity = Vector3.zero
		end

		repeat sleep() until workspace.RuntimeItems:FindFirstChild("Bond")
		tween:Cancel()

		for _,v in pairs(workspace.RuntimeItems:GetChildren()) do
			if v.Name:find("Bond") and v:FindFirstChild("Part") then
				repeat
					sleep()
					if v:FindFirstChild("Part") then
						char.HumanoidRootPart.CFrame = v.Part.CFrame
						srvStorage.Shared.Network.RemotePromise.Remotes.C_ActivateObject:FireServer(v)
					end
				until not v:FindFirstChild("Part")
			end
		end
	end
	sleep()
end
